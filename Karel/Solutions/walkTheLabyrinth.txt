void walkTheLabyrinth() {
    while (!onBeeper()) {
        if (!frontIsClear()) {
            turnRight();
            if (!frontIsClear()) {
                turnLeft();
                turnLeft();
            }
        }
        else {
            moveForward();
        }
    }
}
